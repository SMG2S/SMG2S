143 parVectorMap<S>::parVectorMap(MPI_Comm ncomm, S lbound, S ubound)
144 {
145  MPI_Comm_dup(ncomm, &comm);
146  MPI_Comm_size(comm, &nproc);
147  MPI_Comm_rank(comm, &rank);
148 
149  //setting lower and upper bound
150  lower_bound = lbound;
151  upper_bound = ubound;
152  //get local vector size
153  local_size = upper_bound - lower_bound;
154  //initial global size
155  global_size = 0;
156 
157  MPI_Allreduce(&local_size, &global_size, 1, MPI_INT, MPI_SUM, comm);
158 
159  lprocbound_map.resize(nproc);
160  uprocbound_map.resize(nproc);
161 
162  MPI_Allgather(&lower_bound, 1 , getMPI_Type<S>() , lprocbound_map.data() , 1 , getMPI_Type<S>(), comm ) ;
163  MPI_Allgather(&upper_bound, 1 , getMPI_Type<S>() , uprocbound_map.data(), 1 , getMPI_Type<S>() , comm ) ;
164 
165 }
